#If the Flying fortress is destroyed, but Guardian Angels still stands.
pw_decision_rebuild_flying_fortress = {
	owned_planets_only = yes
	sound = event_construction_02
	icon = pw_decision_rebuild_flying_fortress

	enactment_time = @pw_halfYear

	resources = {
		category = decisions
		cost = {
			influence = @pw_decision_influence_cost
			minerals = @pw_decision_resource_cost
		}
	}

	potential = {
		has_building = pw_building_guardian_angel
		NOT = {
			any_planet_army = {
				army_type = pw_army_flying_fortress
			}
		}
	}

	prerequisites = {
		pw_tech_guardian_angel
	}

	effect = {
		custom_tooltip = pw_decision_rebuild_flying_fortress_tooltip
		create_army = {
			name = pw_army_flying_fortress_name
			owner = THIS.owner
			type = pw_army_flying_fortress
		}
		hidden_effect = {
			planet_event = { id = pw_wonder.1206 }
		}
	}

	ai_weight = {
		weight = 2
		modifier = {
			factor = 0
			planet_devastation > 10
		}
	}
}

#Change Mantle Crucible specialization:
pw_decision_mantle_crucible_alloys = {
	owned_planets_only = yes
	sound = event_structural_collapse
	enactment_time = @pw_1month

	resources = {
		category = decisions
		cost = {
			influence = @pw_decision_influence_cost
			volatile_motes = @pw_building_cost_rare_half
		}
	}

	prerequisites = {
		pw_tech_planetary_mantle_production
	}

	potential = {
		exists = owner
		owner = { country_uses_consumer_goods = yes }
		has_building = pw_building_mantle_crucible
	}

	allow = {
		NOT = { has_modifier = pw_mod_mantle_crucible_alloys }
	}

	effect = {
		if = {
			limit = {
				exists = owner
				owner = { is_machine_empire = no }
			}
			custom_tooltip = pw_decision_mantle_crucible_alloys_regular_tooltip
		}
		else = {
			custom_tooltip = pw_decision_mantle_crucible_alloys_machine_tooltip
		}
		add_modifier = {
			modifier = pw_mod_mantle_crucible_alloys
			days = -1
		}
		hidden_effect = {
			if = {
				limit = { has_modifier = pw_mod_mantle_crucible_consumer_goods }
				remove_modifier = pw_mod_mantle_crucible_consumer_goods
			}
		}
	}

	ai_weight = {
		weight = 2
	}
}
