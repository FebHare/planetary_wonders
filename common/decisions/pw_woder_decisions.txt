	owned_planets_only = yes
	sound = event_structural_collapse

	resources = {
		category = decisions
		cost = {
			influence = @pw_decision_influence_cost
			volatile_motes = @pw_building_cost_rare_half
		}
	}

	prerequisites = {
		pw_tech_planetary_mantle_production
	}

	potential = {
		exists = owner
		owner = { country_uses_consumer_goods = yes }
		has_building = pw_building_mantle_crucible
	}

	allow = {
		NOT = { has_modifier = pw_mod_mantle_crucible_alloys }
	}

	effect = {
		if = {
			limit = {
				exists = owner
				owner = { is_machine_empire = no }
			}
			custom_tooltip = pw_decision_mantle_crucible_alloys_regular_tooltip
		}
		else = {
			custom_tooltip = pw_decision_mantle_crucible_alloys_machine_tooltip
		}
		add_modifier = {
			modifier = pw_mod_mantle_crucible_alloys
			days = -1
		}
		hidden_effect = {
			if = {
				limit = { has_modifier = pw_mod_mantle_crucible_consumer_goods }
				remove_modifier = pw_mod_mantle_crucible_consumer_goods
			}
		}
	}

	ai_weight = {
		weight = 2
	}
}
