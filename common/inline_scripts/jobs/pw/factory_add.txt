triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			$OWNER_CONDITION$
			is_regular_empire = yes
			is_crafter_empire = no
		}
		NOT = { has_modifier = penal_colony }
	}
	job_artisan_add = 1
	mult            = $AMOUNT$
}
